### 5.1.2 导入时重命名

#导入模块的相关方式
import somemodule
from somemodule import somefunction
from somemodule import somedfunction, annotherfunction

# 只有要导入模块中的一切时，才采用最后一种方式
from somemodule import *

import math as foobar
from math import  sqrt as foobar
